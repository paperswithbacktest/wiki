```python
# Example of a simple machine learning model for predicting price movements
import numpy as np
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier

# Sample data: features could be technical indicators, sentiment scores, etc.
X = np.array([[0.5, 1.2, 0.9], [1.1, 0.9, 1.5], [0.3, 1.0, 0.3]])
y = np.array([1, 0, 1])  # Target variable: 1 for price up, 0 for price down

# Split data into training and test sets
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

# Initialize and train the model
model = RandomForestClassifier(n_estimators=100, max_depth=3, random_state=42)
model.fit(X_train, y_train)

# Predicting and evaluating the model
predictions = model.predict(X_test)
print("Predicted outcomes:", predictions)
```
